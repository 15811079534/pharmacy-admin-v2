## 步骤 3: 更新列表页面（5分钟）

### 3.1 修改 script 部分

**完整替换 script 内容：**
```vue
<script setup lang="ts">
import { useTable } from '@/hooks/useTable'
import * as YourApi from '@/api/你的模块/你的文件'
import type { YourPageReqVO } from '@/api/你的模块/你的文件'
import YourForm from './YourForm.vue'

const queryFormRef = ref()
const formRef = ref()

const queryParams = reactive<YourPageReqVO>({
  name: '',
  pageNo: 1,
  pageSize: 10
})

const { loading, tableData, total, getList, handleQuery, handleReset } = useTable({
  fetchData: YourApi.getYourPage
})

const handleAdd = () => {
  formRef.value?.openDialog('create')
}

const handleEdit = (row: any) => {
  formRef.value?.openDialog('update', row.id)
}

const handleDelete = async (row: any) => {
  try {
    await ElMessageBox.confirm('确定要删除吗?', '提示', {
      confirmButtonText: '确定',
      cancelButtonText: '取消',
      type: 'warning'
    })
    await YourApi.deleteYour(row.id)
    ElMessage.success('删除成功')
    await getList()
  } catch (error) {}
}

const handleSuccess = () => {
  getList()
}

onMounted(() => {
  getList()
})
</script>
```

### 3.2 添加表单组件引用

在搜索表单的 `</el-card>` 后面添加：
```vue
<!-- 表单弹窗 -->
<YourForm ref="formRef" @success="handleSuccess" />
```

### 3.3 修改查询按钮
```vue
<el-button @click="() => handleReset(() => queryFormRef.resetFields())">重置</el-button>
```

## 常见问题

### Q: 表单提交后没有刷新列表？
A: 检查是否添加了 `@success="handleSuccess"` 和 `handleSuccess` 方法

### Q: 删除确认弹窗不显示？
A: 检查是否使用了 `try-catch` 包裹

### Q: 图片上传不显示？
A: 检查 `handleBeforeUpload` 和 `handleUpload` 方法是否正确

### Q: 分页不工作？
A: 检查 `queryParams` 是否包含 `pageNo` 和 `pageSize`
